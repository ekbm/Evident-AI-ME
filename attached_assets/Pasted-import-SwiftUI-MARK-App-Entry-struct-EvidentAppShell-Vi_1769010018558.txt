// MARK: - Ask Tab (Hero Screen)

struct AskTab: View {
    @State private var questionText: String = ""
    @State private var isAsking = false
    @State private var lastAnswer: String? = nil

    private let suggestions = [
        "Summarise this document",
        "Find key risks",
        "Explain like I’m 5",
        "Generate a report"
    ]

    var body: some View {
        NavigationStack {
            ScrollView {
                VStack(alignment: .leading, spacing: 16) {

                    // Input Card
                    VStack(alignment: .leading, spacing: 12) {
                        Text("Ask Evident")
                            .font(.title2)
                            .bold()

                        TextField("Ask a question about your documents…", text: $questionText, axis: .vertical)
                            .textFieldStyle(.roundedBorder)
                            .lineLimit(1...4)

                        Button {
                            ask()
                        } label: {
                            HStack {
                                if isAsking { ProgressView().scaleEffect(0.9) }
                                Text(isAsking ? "Thinking…" : "Ask")
                                    .bold()
                                Spacer()
                                Image(systemName: "arrow.up.circle.fill")
                            }
                        }
                        .buttonStyle(.borderedProminent)
                        .disabled(questionText.trimmingCharacters(in: .whitespacesAndNewlines).isEmpty || isAsking)
                    }
                    .padding()
                    .background(.thinMaterial)
                    .clipShape(RoundedRectangle(cornerRadius: 16))

                    // Suggestions
                    VStack(alignment: .leading, spacing: 10) {
                        Text("Quick Suggestions")
                            .font(.headline)
                            .foregroundStyle(.secondary)

                        FlowWrap {
                            ForEach(suggestions, id: \.self) { s in
                                Button {
                                    questionText = s
                                } label: {
                                    Text(s)
                                        .font(.subheadline)
                                }
                                .buttonStyle(.bordered)
                            }
                        }
                    }

                    // Answer
                    if let lastAnswer {
                        VStack(alignment: .leading, spacing: 12) {
                            Text("Answer")
                                .font(.headline)

                            Text(lastAnswer)
                                .font(.body)
                                .frame(maxWidth: .infinity, alignment: .leading)

                            // Post-answer actions (keep v1 clean)
                            Text("Next")
                                .font(.headline)
                                .foregroundStyle(.secondary)

                            FlowWrap {
                                ActionChip(title: "Copy", icon: "doc.on.doc")
                                ActionChip(title: "Export PDF", icon: "square.and.arrow.up")
                                ActionChip(title: "Check Readiness", icon: "checkmark.seal")
                            }
                        }
                        .padding()
                        .background(Color(.secondarySystemBackground))
                        .clipShape(RoundedRectangle(cornerRadius: 16))
                    }

                    Spacer(minLength: 24)
                }
                .padding()
            }
            .navigationTitle("Ask")
            .navigationBarTitleDisplayMode(.inline)
        }
    }

    private func ask() {
        isAsking = true
        let q = questionText.trimmingCharacters(in: .whitespacesAndNewlines)

        // Replace this stub with your API call
        DispatchQueue.main.asyncAfter(deadline: .now() + 0.9) {
            self.lastAnswer = "Here’s a clear answer based on your uploaded documents.\n\nQuestion: “\(q)”\n\n(Stubbed response — wire to your backend.)"
            self.isAsking = false
        }
    }
}

// MARK: - Activity Tab

struct ActivityTab: View {
    @State private var items: [EvidentQAItem] = [
        .init(question: "What are the key risks in the contract?", answerPreview: "Top risks include termination clauses, liability caps…", date: .now),
        .init(question: "Summarise the HR policy", answerPreview: "This policy covers leave, conduct, performance…", date: .now.addingTimeInterval(-3600 * 5)),
        .init(question: "Generate audit summary", answerPreview: "Audit summary: scope, findings, exceptions…", date: .now.addingTimeInterval(-3600 * 26))
    ]

    var body: some View {
        NavigationStack {
            List {
                // Simple grouping: Today / Earlier
                Section("Today") {
                    ForEach(items.filter { Calendar.current.isDateInToday($0.date) }) { item in
                        NavigationLink(value: item) {
                            ActivityRow(item: item)
                        }
                    }
                    if items.filter({ Calendar.current.isDateInToday($0.date) }).isEmpty {
                        Text("No activity today.")
                            .foregroundStyle(.secondary)
                    }
                }

                Section("Earlier") {
                    ForEach(items.filter { !Calendar.current.isDateInToday($0.date) }) { item in
                        NavigationLink(value: item) {
                            ActivityRow(item: item)
                        }
                    }
                    if items.filter({ !Calendar.current.isDateInToday($0.date) }).isEmpty {
                        Text("Nothing yet.")
                            .foregroundStyle(.secondary)
                    }
                }
            }
            .navigationTitle("Activity")
            .navigationDestination(for: EvidentQAItem.self) { item in
                ActivityDetailView(item: item)
            }
        }
    }
}

// MARK: - Small UI Helpers

struct ActionChip: View {
    let title: String
    let icon: String

    var body: some View {
        Button {
            // hook up action
        } label: {
            Label(title, systemImage: icon)
                .font(.subheadline)
        }
        .buttonStyle(.bordered)
    }
}

/// A lightweight flow layout for chips (simple + stable on iOS)
struct FlowWrap<Content: View>: View {
    let content: Content

    init(@ViewBuilder content: () -> Content) {
        self.content = content()
    }

    var body: some View {
        // Basic wrap using LazyVGrid
        LazyVGrid(columns: [GridItem(.adaptive(minimum: 140), spacing: 10)], spacing: 10) {
            content
        }
    }
}
